FigML Examples

This page provides real-world examples of Figma designs exported with FigML and how they translate to code in various frameworks.

Basic Button Component​

FigML XML Output​

<Component name="Button/Primary" type="FRAME" width="120" height="40" cornerRadius="4" fill="#4F46E5">
  <Text name="Label" fontSize="14" fontWeight="500" fill="#FFFFFF" textAlignHorizontal="CENTER" textAlignVertical="CENTER">
    Button Text
  </Text>
</Component>

React Implementation​

import React from 'react';
import styled from 'styled-components';

const StyledButton = styled.button`
  width: 120px;
  height: 40px;
  border-radius: 4px;
  background-color: #4F46E5;
  border: none;
  cursor: pointer;
  display: flex;
  align-items: center;
  justify-content: center;
  
  &:hover {
    background-color: #4338CA;
  }
`;

const ButtonLabel = styled.span`
  font-size: 14px;
  font-weight: 500;
  color: #FFFFFF;
  text-align: center;
`;

const Button = ({ text = "Button Text", onClick }) => {
  return (
    <StyledButton onClick={onClick}>
      <ButtonLabel>{text}</ButtonLabel>
    </StyledButton>
  );
};

export default Button;

React Implementation​

import React from 'react';
import styled from 'styled-components';
import Button from './Button';

const CardContainer = styled.div`
  width: 320px;
  height: 240px;
  background-color: #FFFFFF;
  border-radius: 8px;
  border: 1px solid #EEEEEE;
  display: flex;
  flex-direction: column;
  overflow: hidden;
`;

const CardHeader = styled.div`
  width: 100%;
  height: 48px;
  padding: 16px;
  border-bottom: 1px solid #EEEEEE;
`;

const CardTitle = styled.h3`
  font-size: 16px;
  font-weight: 600;
  margin: 0;
`;

const CardContent = styled.div`
  width: 100%;
  height: 140px;
  padding: 16px;
  display: flex;
  flex-direction: column;
`;

const Description = styled.p`
  font-size: 14px;
  line-height: 20px;
  opacity: 0.8;
  margin: 0;
`;

const Actions = styled.div`
  display: flex;
  flex-direction: row;
  gap: 8px;
  margin-top: 16px;
`;

const Card = ({ title, description, onCancel, onSubmit }) => {
  return (
    <CardContainer>
      <CardHeader>
        <CardTitle>{title}</CardTitle>
      </CardHeader>
      <CardContent>
        <Description>{description}</Description>
        <Actions>
          <Button 
            variant="secondary" 
            text="Cancel" 
            onClick={onCancel} 
          />
          <Button 
            variant="primary" 
            text="Submit" 
            onClick={onSubmit} 
          />
        </Actions>
      </CardContent>
    </CardContainer>
  );
};

export default Card;

React Implementation with Tailwind CSS​

import React, { useState } from 'react';

const LoginForm = ({ onSubmit }) => {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [rememberMe, setRememberMe] = useState(false);
  
  const handleSubmit = (e) => {
    e.preventDefault();
    onSubmit({ email, password, rememberMe });
  };
  
  return (
    <div className="w-[400px] bg-white rounded-lg p-6 shadow-md">
      <h2 className="text-2xl font-semibold mb-6">Log In</h2>
      
      <form onSubmit={handleSubmit}>
        <div className="mb-4">
          <label 
            htmlFor="email" 
            className="block text-sm font-medium mb-2"
          >
            Email
          </label>
          <input
            id="email"
            type="email"
            className="w-full h-10 px-3 py-2 bg-gray-100 border border-gray-200 rounded-md focus:outline-none focus:ring-2 focus:ring-indigo-500"
            value={email}
            onChange={(e) => setEmail(e.target.value)}
            required
          />
        </div>
        
        <div className="mb-4">
          <label 
            htmlFor="password" 
            className="block text-sm font-medium mb-2"
          >
            Password
          </label>
          <input
            id="password"
            type="password"
            className="w-full h-10 px-3 py-2 bg-gray-100 border border-gray-200 rounded-md focus:outline-none focus:ring-2 focus:ring-indigo-500"
            value={password}
            onChange={(e) => setPassword(e.target.value)}
            required
          />
        </div>
        
        <div className="flex items-center mb-6">
          <input
            id="remember-me"
            type="checkbox"
            className="w-4 h-4 text-indigo-600 border-gray-300 rounded focus:ring-indigo-500"
            checked={rememberMe}
            onChange={(e) => setRememberMe(e.target.checked)}
          />
          <label 
            htmlFor="remember-me" 
            className="ml-2 block text-sm text-gray-700"
          >
            Remember me
          </label>
        </div>
        
        <button
          type="submit"
          className="w-full h-12 bg-indigo-600 text-white font-medium rounded-md hover:bg-indigo-700 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-indigo-500"
        >
          Log In
        </button>
      </form>
    </div>
  );
};

export default LoginForm;
